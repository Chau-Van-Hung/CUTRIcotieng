function speakCurrent(){ // dừng nếu đang đọc người trước window.speechSynthesis.cancel(); const text = document.getElementById("data").innerText; const speech = new SpeechSynthesisUtterance(text); speech.lang = "vi-VN"; speech.rate = 0.9; // chậm trang trọng speech.pitch = 1.15; // giọng nữ dịu const voices = window.speechSynthesis.getVoices(); // tìm giọng tiếng Việt const femaleVoice = voices.find(v => v.lang.includes("vi")); if(femaleVoice){ speech.voice = femaleVoice; } window.speechSynthesis.speak(speech); } // tránh lỗi chưa load giọng window.speechSynthesis.onvoiceschanged = () => {}; // tránh bị chặn autoplay document.body.addEventListener("click", () => { window.speechSynthesis.resume(); });
